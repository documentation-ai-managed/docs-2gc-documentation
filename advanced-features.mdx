<Tabs>
  <Tab title="Webhooks" icon="webhook">
    Set up webhooks to trigger actions when documents change.
    
    <CodeGroup tabs="Setup,Payload">
      ```javascript
      // Register a webhook
      const registerWebhook = async (url, events) => {
        const response = await fetch('/api/webhooks', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ url, events })
        });
        return response.json();
      };
      ```
      ```json
      {
        "event": "document.updated",
        "data": {
          "id": "doc123",
          "title": "Updated Document",
          "timestamp": "2024-01-15T10:30:00Z"
        }
      }
      ```
    </CodeGroup>

<Expandable title="Git Integration" default-open="false">
  Connect your documentation to Git repositories for version control and collaboration.
  
  <Request tabs="Sync Repository" show-lines="true">
    ```bash
    curl -X POST https://api.2gc.com/git/sync \
      -H "Authorization: Bearer YOUR_TOKEN" \
      -d '{"repo": "https://github.com/user/repo"}'
    ```
  </Request>
</Expandable>

<Request tabs="Test Integration" show-lines="false">
  ```javascript
  const testIntegration = async (service, config) => {
    const response = await fetch(`/api/integrations/${service}/test`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(config)
    });
    return response.json();
  };
  ```
</Request>
